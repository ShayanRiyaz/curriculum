class Node:
    def __init__(self, data):

        self.left = None
        self.right = None
        self.data = data
        
    def inorderTraversal(root):
        res, stack = [], []
        while True:
            while root:
                stack.append(root)
                root = root.left
            if not stack:
                return res
            node = stack.pop()
            res.append(node.data)
            print(node.data, end=" ") 
            root = node.right
